if CLIENT then
    
    local RenderLib = loadLibrary("render")

    function ChBox1(State)
        if State == true then
        RenderLib.setColor(255,0,0,255)
        elseif State == false then
        RenderLib.setColor(0,0,255,255)
        end
        RenderLib.drawRectOutline(444, 10, 60, 60)
    end
